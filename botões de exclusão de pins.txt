HTML: Já está presente no HTML:
JS: 

let currentPrinterIndex = null; // controla a impressora aberta no modal

function showModal(printer, index) {
    currentPrinterIndex = index; // salva índice atual
    mTitle.textContent = printer.title;
    mDesc.textContent = printer.desc;
    mModel.textContent = printer.model;
    mSerial.textContent = printer.serial;
    mLoc.textContent = printer.loc;
    mNotes.textContent = printer.notes;
    mImg.src = printer.img;
    modal.style.display = 'flex';
}

// Botão de deletar impressora dentro do modal
document.getElementById("deletePrinterBtn").addEventListener("click", () => {
    if (currentPrinterIndex !== null) {
        printers.splice(currentPrinterIndex, 1); // remove do array
        renderPins(); // atualiza o mapa
        modal.style.display = 'none'; // fecha modal
        currentPrinterIndex = null;
    }
});

JS:

const deleteAllBtn = document.getElementById('deleteAllBtn');
const confirmDeleteAll = document.getElementById('confirmDeleteAll');
const confirmYes = document.getElementById('confirmYes');
const confirmNo = document.getElementById('confirmNo');

deleteAllBtn.addEventListener('click', () => {
    confirmDeleteAll.style.display = 'inline';
});

confirmYes.addEventListener('click', () => {
    printers.length = 0; // limpa array
    renderPins();
    confirmDeleteAll.style.display = 'none';
});

confirmNo.addEventListener('click', () => {
    confirmDeleteAll.style.display = 'none';
});
